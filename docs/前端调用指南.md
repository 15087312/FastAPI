## 📡 获取服务配置

### **动态获取服务地址（推荐）**

```javascript
// 先获取配置信息
async function getServiceConfig() {
    const response = await fetch('http://localhost:8000/config');
    const config = await response.json();
    
    return {
        baseUrl: config.api_base_url,  // http://localhost:8000/api/v1
        healthUrl: config.health_url,  // /health
        docsUrl: config.docs_url       // /docs
    };
}

// 使用示例
const config = await getServiceConfig();
console.log('API 地址:', config.baseUrl);
```

```json

  "port": 8000,
  "api_base_url": "http://localhost:8000/api/v1",
  "docs_url": "/docs",
  "health_url": "/health",
  "debug": false
}
```

## 💻 完整示例代码

### **Vue 3 示例**

```vue
<template>
  <div>
    <h1>库存管理</h1>
    <button @click="checkService">检查服务</button>
    <button @click="queryStock">查询库存</button>
    <button @click="reserveStock">预占库存</button>
    
    <div v-if="result">
      <pre>{{ result }}</pre>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue';

const API_BASE = 'http://localhost:8000/api/v1';
const result = ref(null);

// 检查服务
async function checkService() {
    try {
        const res = await fetch('http://localhost:8000/health');
        result.value = await res.json();
    } catch (error) {
        result.value = { error: error.message };
    }
}

// 查询库存
async function queryStock() {
    try {
        const res = await fetch(`${API_BASE}/stock?product_id=1`);
        result.value = await res.json();
    } catch (error) {
        result.value = { error: error.message };
    }
}

// 预占库存
async function reserveStock() {
    try {
        const res = await fetch(`${API_BASE}/reserve`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                product_id: 1,
                quantity: 10,
                order_id: 'ORDER_' + Date.now()
            })
        });
        result.value = await res.json();
    } catch (error) {
        result.value = { error: error.message };
    }
}
</script>
```

### **React 示例**

```jsx
import { useState, useEffect } from 'react';

function InventoryApp() {
    const [config, setConfig] = useState(null);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState(null);

    useEffect(() => {
        // 获取服务配置
        fetch('http://localhost:8000/config')
            .then(res => res.json())
            .then(data => setConfig(data))
            .catch(err => setError(err.message));
    }, []);

    const handleReserve = async () => {
        setLoading(true);
        try {
            const res = await fetch(`${config.api_base_url}/reserve`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    product_id: 1,
                    quantity: 10,
                    order_id: 'ORDER_' + Date.now()
                })
            });
            const data = await res.json();
            console.log(data);
        } catch (err) {
            setError(err.message);
        } finally {
            setLoading(false);
        }
    };

    if (!config) return <div>加载中...</div>;

    return (
        <div>
            <h1>库存服务</h1>
            <p>API 地址：{config.api_base_url}</p>
            <button onClick={handleReserve} disabled={loading}>
                {loading ? '处理中...' : '预占库存'}
            </button>
            {error && <p style={{color: 'red'}}>{error}</p>}
        </div>
    );
}
```

### **原生 JavaScript 示例**

```javascript
class InventoryClient {
    constructor(baseUrl = 'http://localhost:8000/api/v1') {
        this.baseUrl = baseUrl;
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseUrl}${endpoint}`;
        const config = {
            headers: {
                'Content-Type': 'application/json',
                ...options.headers
            },
            ...options
        };

        const response = await fetch(url, config);
        const data = await response.json();

        if (!response.ok) {
            throw new Error(data.message || '请求失败');
        }

        return data;
    }

    // 健康检查
    async health() {
        return this.request('/health');
    }

    // 查询库存
    async getStock(productId) {
        return this.request(`/stock?product_id=${productId}`);
    }

    // 预占库存
    async reserve(productId, quantity, orderId) {
        return this.request('/reserve', {
            method: 'POST',
            body: JSON.stringify({ product_id: productId, quantity, order_id: orderId })
        });
    }

    // 确认库存
    async confirm(orderId) {
        return this.request(`/confirm/${orderId}`, { method: 'POST' });
    }

    // 释放库存
    async release(orderId) {
        return this.request(`/release/${orderId}`, { method: 'POST' });
    }
}

// 使用示例
const client = new InventoryClient();

client.health().then(console.log);
client.getStock(1).then(console.log);
client.reserve(1, 10, 'ORDER_001').then(console.log);
```

## ⚠️ 注意事项

### **1. CORS 跨域**

后端已启用 CORS，允许所有来源访问（开发环境）：
```python
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # 生产环境改为具体域名
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
```

### **2. 错误处理**

```javascript
try {
    const response = await fetch('http://localhost:8000/api/v1/reserve', {
        method: 'POST',
        body: JSON.stringify({...})
    });
    
    const data = await response.json();
    
    if (!response.ok) {
        console.error('错误:', data.message);
        return;
    }
    
    console.log('成功:', data);
} catch (error) {
    console.error('网络错误:', error);
}
```

### **3. 状态码说明**

| 状态码 | 说明 |
|--------|------|
| 200 | 请求成功 |
| 400 | 请求参数错误 |
| 404 | 资源未找到 |
| 422 | 请求验证失败 |
| 429 | 请求过于频繁 |
| 500 | 服务器内部错误 |
